142  def __emit_aixh_node(self, ir_node: AxfcIRNode) -> {AxfcError, AIXLayer}:
143  # logging.info("AxfcTFIRTranslator:_emit_aixh_node - node %d", ir_node.layer_id)
144 
145  # get the operation information specified in the machine description
146  layer_info = self._md.get_layer_info(ir_node.op)
147 
148  # create a new AIX layer
149  aix_layer = AIXLayer()
150 
151  # layer ID
152  aix_layer.id = ir_node.layer_id
153  aix_layer.name = ir_node.name
154 
155  # layer types
156  layer_type = AIXLayer.AIXLayerType.Value(layer_info.layer)
157  aix_layer.type.append(layer_type)
158 
159  # emit the output specific to each AIX layer type
160  try:
161  emit_aix_layer = self.__emit_aix_layer_tbl[layer_type]
162  except KeyError:
163  logging.warning("__emit_aixh_node: unsupported layer type - %s, %s", ir_node.op, layer_type)
164  return AxfcError.UNSUPPORTED_AIX_LAYER_EMIT, None
165 
166  # register the generated AIX layer
167  ir_node.aix_layer = aix_layer
168 
169  # perform the emission of AIX layers
170  err = emit_aix_layer(ir_node)
171 
172  if err is not AxfcError.SUCCESS:
173  logging.warning("AxfcIRTranslator:__emit_aixh_node - %s", err)
174  return err, None
175 
176  # layer input and output
177  if ir_node.is_input:
178  aix_layer.type.append(AIXLayer.AIXLayerType.AIX_LAYER_INPUT)
179 
180  if ir_node.is_output:
181  aix_layer.type.append(AIXLayer.AIXLayerType.AIX_LAYER_OUTPUT)
182 
183  # predecessors & successors
184  for pred in ir_node.preds:
185  if pred.is_aixh_support:
186  aix_layer.preds.append(pred.layer_id)
187 
188  for succ in ir_node.succs:
189  if succ.is_aixh_support:
190  aix_layer.succs.append(succ.layer_id)
191 
192  # activation
193  activation = layer_info.activation
194  if activation is not None:
195  aix_layer.activation = AIXLayer.AIXActivationMode.Value(activation)
196 
197  return AxfcError.SUCCESS, aix_layer
